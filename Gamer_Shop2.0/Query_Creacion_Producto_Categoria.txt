CREATE DATABASE ProyectoTallerII;

USE ProyectoTallerII;

CREATE TABLE Categoría_producto
(
  ID_Categoria INT NOT NULL,
  Nombre_Categoria VARCHAR(50) NOT NULL,
  CONSTRAINT PK_Categoria PRIMARY KEY (ID_Categoria)
);

CREATE TABLE Producto
(
  ID_Producto INT NOT NULL IDENTITY,
  Serial INT NOT NULL,
  Nombre VARCHAR(100) NOT NULL,
  Descripcion VARCHAR(200) NOT NULL,
  Stock INT NOT NULL,
  Precio FLOAT NOT NULL,
  ID_Categoria INT NOT NULL,
  CONSTRAINT PK_Producto PRIMARY KEY (ID_Producto, Serial),
  CONSTRAINT FK_CategoriaProducto FOREIGN KEY (ID_Categoria) REFERENCES Categoría_producto(ID_Categoria)
);
