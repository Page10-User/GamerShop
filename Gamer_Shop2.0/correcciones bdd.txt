ALTER TABLE Categoría_producto ADD ID_Categoria INT NOT NULL IDENTITY;

ALTER TABLE Categoría_producto ADD CONSTRAINT FK_CategoriaProducto FOREIGN KEY (ID_Categoria) REFERENCES Categoría_producto(ID_Categoria);

-- Creacion tabla proveedor 
CREATE TABLE Proveedor
(
  ID_Proveedor INT NOT NULL IDENTITY,
  Razon_social VARCHAR(35) NOT NULL,
  Nombre_representante VARCHAR(35) NOT NULL,
  Telefono VARCHAR(15) NOT NULL,
  Correo VARCHAR(50) NOT NULL,
  Dirección VARCHAR(150) NOT NULL,
  CONSTRAINT PK_Proveedor PRIMARY KEY (ID_Proveedor),
  CONSTRAINT UQ_Razon_social UNIQUE (Razon_social),
  CONSTRAINT UQ_Telefono UNIQUE (Telefono),
  CONSTRAINT UQ_Correo UNIQUE (Correo),
  CONSTRAINT UQ_Dirección UNIQUE (Dirección)
);
